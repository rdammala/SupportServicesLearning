interface Refund {
  id: string;
  amount: number;
  status: 'pending' | 'approved' | 'rejected';
}

  const [refunds, setRefunds] = useState([]);
  const [loading, setLoading] = useState(false);

    const fetchRefunds = async () => {
      setLoading(true);
      const response = await fetch('/api/refunds');
      const data = await response.json();
