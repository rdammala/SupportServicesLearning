🔵 C# / .NET 10

Async HttpClient Call
public async Task GetAsync(string url)
{
    var response = await _httpClient.GetAsync(url);
    response.EnsureSuccessStatusCode();
    return await response.Content.ReadAsAsync();
}
Dependency Injection Registration
services.AddScoped();
services.AddHttpClient(client =>
{
    client.BaseAddress = new Uri("https://api.example.com");
});
services.AddOptions()
    .BindConfiguration("AppSettings")
    .ValidateDataAnnotations();

var results = _refunds

📊 KQL Queries

Trace Single Request
union requests, dependencies, exceptions, traces
| where operation_Id == "your-correlation-id"
| project timestamp, itemType, name, success, resultCode
| order by timestamp asc
Error Rate by Operation
requests
| where timestamp > ago(1h)
| summarize failed = sum(1 - toint(success)),
            total = count() by name
| extend error_rate = 100.0 * failed / total
| order by error_rate desc
P95 Latency Timechart
requests
| where timestamp > ago(24h)
| summarize p95_latency = percentile(duration, 95) by bin(timestamp, 5m)
| render timechart
Scope: Affected User Count
exceptions
| where timestamp > ago(1h)
| summarize affected_users = dcount(tostring(customDimensions.userId)),
            error_count = count() by type
| order by error_count desc

🏗️ Bicep / IaC

Storage Account
param location string = 'eastus'
param environment string = 'dev'

resource storageAccount 'Microsoft.Storage/storageAccounts@2021-04-01' = {
  name: 'st${environment}${uniqueString(resourceGroup().id)}'
  location: location
  kind: 'StorageV2'
  sku: {
    name: 'Standard_LRS'
  }
}

output storageId string = storageAccount.id

  --resource-group my-rg \
  --template-file main.bicep \
  --parameters environment=prod location=eastus
Query Cosmos DB
az cosmosdb sql query \
  --account-name myaccount \
  --database-name mydb \
  --container-name mycontainer \
  --query 'SELECT * FROM c WHERE c.id = @id' \
  --parameters '@id=12345'
View App Insights Logs
az monitor app-insights metrics show \
  --app myappinsights \
  --metric requests/count \
  --aggregation total \
  --interval PT1H

Multi-Stage Pipeline
trigger:
  - main

stages:
  - stage: Build
    jobs:
      - job: BuildJob
        steps:
          - script: dotnet build
          - script: dotnet test

  - stage: Deploy
    condition: succeeded()
    jobs:
      - deployment: DeployProd
        environment: production
        strategy:
          runOnce:
            deploy:
              - script: ./deploy.sh
